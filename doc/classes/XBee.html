# File lib/ruby_xbee.rb, line 50
def getresults( sp, echo = true )
  results = ""
  while (c = sp.getc) do
    if ( !echo.nil? && echo )
      putc c
    end
    results += "#{c.chr}"
  end

  # deal with multiple lines
  results.gsub!( "\r", "\n")
end
